#!/bin/bash

# Get the directory where this script is located
DIR="$( cd "$( dirname "${BASH_SOURCE[0]}" )" && pwd )"

# Run Java
# Note: We use 'python3' here because that is standard on Mac
# We point to the MAC version of the JavaFX SDK
java --module-path "$DIR/javafx-sdk-mac/lib" --add-modules javafx.controls,javafx.fxml -jar "$DIR/backgammonJarTry1.jar"
